# aloop Pi4 boot config additions (appended to config.txt / usercfg.txt).
# Put the USB-C/OTG port into peripheral (gadget) mode so f_uac2 can present the
# Pi as a USB audio device to a host.
dtoverlay=dwc2,dr_mode=peripheral

# Serial console on the GPIO UART (GPIO14 TX / GPIO15 RX, 115200 8N1) so first
# boot + bring-up is debuggable from a card WITHOUT an HDMI display — essential
# for testing. See docs/FLASHING.md for the pinout.
enable_uart=1
